import cv2 as cv
import numpy as np

# set up the image list to store images
imgs=[]
with zipfile.ZipFile('../content/drive/My Drive/small_img.zip') as zip:
    infos = zip.infolist()
    for img in infos:
        imgs.append([Image.open(zip.open(img.filename))])

# test if the inputed word in the images
txt='Christopher'  # User input the word to be searched  
text=[]
check_if_in=[]
for i in range(len(imgs)):
    text.append(pytesseract.image_to_string(imgs[i][0]))
    if txt in text[i]:
        check_if_in.append(i)

    
# loading the face detection classifier
face_cascade = cv.CascadeClassifier('../content/drive/My Drive/haarcascade_frontalface_default.xml')

# set up the image list to store images
imgs_cv=[]
with zipfile.ZipFile('../content/drive/My Drive/small_img.zip') as zip:
    for img in zip.infolist():
        data = zip.read(img.filename)
        cvimg = cv.imdecode(np.frombuffer(data, np.uint8), 1)
        imgs_cv.append([cvimg])


# select the face in each picture
for i in range(len(imgs_cv)):
    faces = face_cascade.detectMultiScale(imgs_cv[i][0], 1.35)
    imgs[i].append([faces])

# crop the img with bounding box and svae to thumbnails size
size = 128, 128
for i in range(len(imgs)):
    thumbs=[]
    for arr in imgs[i][1]:
        for x, y, w, h in arr:
            im = imgs[i][0].crop((x, y, x+w, y+h)).copy()
            im.thumbnail(size)
            thumbs.append(im)
            im = None
    imgs[i].append(thumbs)

# create a contact sheet list stored for the selected picture
result=[]
thumb_width = 128 
for i in range(len(imgs)):
    if type(imgs[i][1][0])=='tuple':
        result.append('')
        continue
    else:        
        contact_sheet=PIL.Image.new('RGB', (128*5,128*(len(imgs[i][2])//5+1)))
        x, y=0, 0
        for img in imgs[i][2]:
            contact_sheet.paste(img, (x, y))
            if x+thumb_width == contact_sheet.width:
                x=0
                y=y+thumb_width
            else:
                x=x+thumb_width
        result.append(contact_sheet)

# show the contact sheet of the page where contains the word inputed
